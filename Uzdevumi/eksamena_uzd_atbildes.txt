        if summa > 0: # P캐rbaud캐m, vai summa ir pozit캶va
            self.bilance += summa
            print(f"Ieskait캶ti {summa:.2f} EUR. Jaun캐 bilance: {self.bilance:.2f} EUR")
        else:
            print("K컆콞da: Ieskait캐m캐 summa nevar b콞t negat캶va vai nulle.")

    # Metode, lai iz켽emtu naudu no konta

        # P캐rbaud캐m, vai kont캐 ir pietiekami daudz naudas

            self.bilance -= summa
            print(f"Iz켽emti {summa:.2f} EUR. Jaun캐 bilance: {self.bilance:.2f} EUR")
        elif summa <= 0:
            print("K컆콞da: Iz켽emam캐 summa nevar b콞t negat캶va vai nulle.")
        else:
            print(f"K컆콞da: Nepietiek l캶dzek컆u! J콞su kont캐 ir tikai {self.bilance:.2f} EUR.")
